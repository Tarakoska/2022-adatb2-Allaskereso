CREATE SCHEMA AUTHORIZATION
CREATE TABLE Szakmak(
id NUMBER GENERATED BY DEFAULT AS IDENTITY,
nev VARCHAR(50),
leiras VARCHAR(255),
CONSTRAINT szakmak_pk PRIMARY KEY(id)
);

CREATE TABLE Felhasznalo(
id NUMBER GENERATED BY DEFAULT AS IDENTITY,
veznev VARCHAR(50) NOT NULL,
kernev VARCHAR(50) NOT NULL,
felhnev VARCHAR(50) NOT NULL,
jelszo VARCHAR(50) NOT NULL,
email VARCHAR(50),
varos VARCHAR(50) NOT NULL,
utca VARCHAR(50) NOT NULL,
hazszam NUMBER NOT NULL,
telefon VARCHAR(50) NOT NULL,
isadmin NUMBER,
CONSTRAINT felhasznalo_pk PRIMARY KEY(id)
);


CREATE TABLE Oneletrajzok(
id NUMBER GENERATED BY DEFAULT AS IDENTITY,
filenev VARCHAR(50),
formatum VARCHAR(25),
feid NUMBER,
CONSTRAINT oneletrajzok_pk PRIMARY KEY(id),
CONSTRAINT felhid_fk_in_oneletrajzok FOREIGN KEY(feid) REFERENCES Felhasznalo (id)
);

CREATE TABLE Hirdetesek(
id NUMBER GENERATED BY DEFAULT AS IDENTITY,
nev VARCHAR(50),
leiras VARCHAR(255),
szakid NUMBER,
CONSTRAINT hirdetesek_pk PRIMARY KEY(id),
CONSTRAINT felhid_fk_in_szakmak FOREIGN KEY(szakid) REFERENCES Szakmak (id)
);

CREATE TABLE Munkaado(
id NUMBER GENERATED BY DEFAULT AS IDENTITY,
beosztas VARCHAR(50),
ertekeles NUMBER,
feid NUMBER,
CONSTRAINT munkaado_pk PRIMARY KEY(id),
CONSTRAINT felhid_fk_in_munkaado FOREIGN KEY(feid) REFERENCES Felhasznalo (id)
);

CREATE TABLE HirdetesFeladas(
id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY,
mikor TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
muid NUMBER,
hiid NUMBER,
CONSTRAINT hirdetesfeladas_pk PRIMARY KEY(id),
CONSTRAINT munid_fk_in_hirdetesfeladas FOREIGN KEY(muid) REFERENCES Munkaado (id),
CONSTRAINT hirid_fk_in_hirdetesfeladas FOREIGN KEY(hiid) REFERENCES Hirdetesek (id)
);

CREATE TABLE Allaskereso(
id NUMBER GENERATED BY DEFAULT AS IDENTITY,
szulido DATE NOT NULL,
onid NUMBER,
feid NUMBER,
CONSTRAINT allaskereso_pk PRIMARY KEY(id),
CONSTRAINT oneletid_fk_in_allaskereso FOREIGN KEY(onid) REFERENCES Oneletrajzok (id),
CONSTRAINT felh_fk_in_allaskereso FOREIGN KEY(feid) REFERENCES Felhasznalo (id)
);

CREATE TABLE Birtokol(
id NUMBER GENERATED BY DEFAULT AS IDENTITY,
szakid NUMBER,
allid NUMBER,
CONSTRAINT birtokol_pk PRIMARY KEY(id),
CONSTRAINT allid_fk_in_birtokol FOREIGN KEY(allid) REFERENCES Allaskereso (id),
CONSTRAINT szakmakid_fk_in_birtokol FOREIGN KEY(szakid) REFERENCES Szakmak (id)
);

CREATE TABLE Jelentkezes(
id NUMBER GENERATED BY DEFAULT AS IDENTITY,
mikor TIMESTAMP WITH TIME ZONE NOT NULL,
allid NUMBER,
hiid NUMBER,
CONSTRAINT jelentkezes_pk PRIMARY KEY(id),
CONSTRAINT allid_fk_in_jelentkezes FOREIGN KEY(allid) REFERENCES Allaskereso (id),
CONSTRAINT hirid_fk_in_jelentkezes FOREIGN KEY(hiid) REFERENCES Hirdetesek (id)
);

CREATE OR REPLACE TRIGGER azonos_fh
BEFORE INSERT OR UPDATE
   ON Felhasznalo
   FOR EACH ROW
DECLARE
   fh_count number;
BEGIN
   SELECT COUNT(*) INTO fh_count FROM Felhasznalo WHERE FELHNEV = :new.FELHNEV;
   if fh_count > 0 then
      RAISE_APPLICATION_ERROR(-20002,'Már létezik ilyen felhasználónév!');
   end if;
END;

CREATE OR REPLACE TRIGGER felhasznalo_email
BEFORE INSERT OR UPDATE
   ON FELHASZNALO
   FOR EACH ROW
DECLARE
   formatum NUMBER;
BEGIN
   SELECT COUNT(*) INTO formatum FROM Felhasznalo WHERE :NEW.EMAIL LIKE '%@%.%' AND REGEXP_COUNT(:NEW.EMAIL, '@') = 1;
   if formatum = 0 then
      RAISE_APPLICATION_ERROR(-20003,'Érvénytelen email formátum');
   end if;
END;

CREATE OR REPLACE TRIGGER delete_admin
BEFORE DELETE ON Felhasznalo
   FOR EACH ROW
   WHEN (OLD.ISADMIN = 1)
BEGIN
      RAISE_APPLICATION_ERROR(-20005,'Admint nem lehet törölni!');
END;

-- 24 órán belül jelentkezett valamilyen állásra álláskeresők neve
SELECT (VEZNEV || ' ' || KERNEV) as NEV FROM ((ALLASKERESO A
    INNER JOIN FELHASZNALO F ON F.ID = A.FEID)
    INNER JOIN JELENTKEZES J ON J.ALLID = A.ID)
    WHERE J.MIKOR > CURRENT_TIMESTAMP - interval '1' day;

-- Munkaadók felhasználóneve legalább 2 hírdetéssel
SELECT FELHNEV FROM (SELECT FELHNEV, COUNT(H.ID) as darab FROM ((MUNKAADO M
    INNER JOIN FELHASZNALO F on F.ID = M.FEID)
    LEFT JOIN HIRDETESFELADAS H ON H.MUID = M.ID)
    GROUP BY FELHNEV) WHERE darab > 1;

-- Álláskeresők neve legalább 1 szakmával
SELECT NEV FROM (SELECT (F.VEZNEV || ' ' || F.KERNEV) as NEV, COUNT(B.ALLID) as darab
    FROM ((ALLASKERESO A
    INNER JOIN FELHASZNALO F on F.ID = A.FEID)
    LEFT JOIN BIRTOKOL B on B.ALLID = A.ID)
    GROUP BY VEZNEV, KERNEV) WHERE darab > 0;

-- Legnépszerűbb szakmák
SELECT NEV, COUNT(B.ALLID) as DARAB
    FROM SZAKMAK SZ INNER JOIN BIRTOKOL B on SZ.ID = B.SZAKID
    GROUP BY NEV;

-- Felhasználónak megfelelő szakmák. KELL BELE ID
SELECT H.ID,H.NEV FROM HIRDETESEK H
    INNER JOIN SZAKMAK SZ ON H.SZAKID = SZ.ID
    INNER JOIN BIRTOKOL B on SZ.ID = B.SZAKID
    INNER JOIN ALLASKERESO A on B.ALLID = A.ID
    WHERE A.FEID = ':id';

-- Hírdetés a legtöbb jelentkezővel
SELECT * FROM (
    SELECT NEV, LEIRAS, MAX(J.ID) as JELENTKEZOK
    FROM HIRDETESEK H
    INNER JOIN JELENTKEZES J on H.ID = J.HIID
    GROUP BY NEV, LEIRAS
) ORDER BY JELENTKEZOK DESC;

-- A legfiatalabb álláskereső
SELECT VEZNEV, KERNEV, FELHNEV, EMAIL, VAROS, UTCA, HAZSZAM, TELEFON, MAX(A.SZULIDO) as SZULETESI_IDO FROM ALLASKERESO A
    INNER JOIN FELHASZNALO F on A.FEID = F.ID
    WHERE SZULIDO = (SELECT MAX(SZULIDO) from ALLASKERESO)
    GROUP BY VEZNEV, KERNEV, FELHNEV, EMAIL, VAROS, UTCA, HAZSZAM, TELEFON;

-- Legelső hirdetők
SELECT VEZNEV, KERNEV, FELHNEV, JELSZO, EMAIL, VAROS, UTCA, HAZSZAM, TELEFON, ISADMIN FROM FELHASZNALO
Inner join MUNKAADO on FELHASZNALO.ID = FEID
Inner join HIRDETESFELADAS on MUNKAADO.ID = MUID
WHERE HIRDETESFELADAS.MIKOR = (SELECT MIN(HIRDETESFELADAS.MIKOR) FROM HIRDETESFELADAS)
ORDER BY VEZNEV;
